```Cpp HL:3,11-17,19 title:a
#include <iostream>
#include <string>
#include <vector>

using namespace std;

class Solution {
   public:
    // note: 此函数判断一个由int转化过来的string是否是单调递增的
    bool isRaising(string str) {
        for (int i = 0; i < str.size() - 1; i++) {
            if (str[i] > str[i + 1]) {
                return false;
            }
        }
        return true;
    }
    // note: 对string的第i位进行判别，看是否从这一位开始有问题
    // 输入("12533",2);则会判断"12533"与"12555"的大小
    // 原str"12533"比"12555"小，说明从这一位开始有问题，于是将5减1，并将后面的所有位都改为9
    // 得到"12499"
    bool utils(string& str, int i) {
        int size = str.size();

        string temp = str;
        for (int j = i; j < size; j++) {
            temp[j] = str[i];
        }
        if (str < temp) {
            str[i] = str[i] - 1;
            for (int j = i + 1; j < size; j++) {
                str[j] = '9';
            }
            return false;
        }
        return true;
    }
    int monotoneIncreasingDigits(int n) {
        string str = to_string(n);
        if (isRaising(str)) {
            return n;
        }
        for (int i = 0; i < str.size(); i++) {
            if (!utils(str, i)) {
                int a = stoi(str);
                return a;
            }
        }
        return 0;
    }
};

int main() {
    Solution sol;
    int result = sol.monotoneIncreasingDigits(1001);
    cout << result << endl;
}
```
